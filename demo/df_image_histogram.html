def plot_segment(image, mask, alpha = 0.5):
    '''
    image: np.array of size [224,224,3]
    mask: np.array of size [224,224,3]
    alpha: alpha of the histogram plot
    '''
    seg = np.multiply(image, mask > 0)
    channel = ["red", "green", "blue"]
    
    for i in range(len(channel)):
        df = seg[:,:,i].flatten()
        plt.hist(df[df > 0], density = True, bins = 50, histtype = "stepfilled",
                 color = channel[i], alpha = alpha)
    plt.show()
